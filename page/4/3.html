                    // GSAP TIMELINE 클래스 여러가지를 한번에 이용해서..
                    gsap.utils.toArray(".ani_tb").forEach(ani_tb => {
                        var tl = gsap.timeline({
                            scrollTrigger: {
                                trigger: ani_tb,
                                start: "top 100%"
                            }
                        });
                        tl.to(ani_tb, { duration: 1.5, opacity: 1, y: 0 });
                    });
